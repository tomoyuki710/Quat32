Usage

Vector3 axis;
float angle;
Quaternion quaternion = Quaternion.AngleAxis(angle, axis);

// Compress (Quaternion -> Quat32 -> int)
Quat32 quat32 = quaternion;
int compressed = quat32;

// Decompress (int -> Quat32 -> Quaternion)
